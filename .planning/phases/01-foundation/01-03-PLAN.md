<objective>
Build the operator frontend — React 19 SPA with login screen, status bar showing live hardware health, and command panel for OBS scene switching and PTZ preset recall/save. Every command uses optimistic UI with ACK-based state transitions (pending → confirmed → error). The status bar updates in real-time via WebSocket from the cloud backend.

Purpose: The operator needs a visual interface to log in, see system health, and control cameras. The ACK-based UX builds trust — the UI never lies about hardware state.
Output: Working SPA (Vite + React 19 + Tailwind CSS 4) deployable alongside the cloud backend on Railway.
</objective>

REST Endpoints:
```
POST /api/auth/login  { passphrase: string }  →  { token: string }  |  401 { error: string }
GET  /api/health       (Authorization: Bearer <token>)  →  { status: 'ok', user: {...} }
POST /api/obs/scene    { sceneName: string }  →  202 { requestId, status: 'pending' }
POST /api/ptz/preset/recall  { presetNumber: number (1-8) }  →  202 { requestId, status: 'pending' }
POST /api/ptz/preset/save    { presetNumber: number (1-8) }  →  202 { requestId, status: 'pending' }
GET  /api/agent/status  →  { connected, lastHeartbeat, health: { obs, ptz } }
GET  /api/command/:requestId  →  { requestId, status: 'pending'|'ok'|'error'|'stale', error? }
```

WebSocket (/frontend?token=<jwt>):
```
Server → Client: { type: 'agent_health', subsystems: { obs: boolean, ptz: boolean } }
Server → Client: { type: 'command_result', requestId: string, status: 'ok'|'error'|'stale', error?: string }
```

    **Step 1 — WebSocket Hook (`cloud/client/src/hooks/useWebSocket.ts`):**
    Create a custom hook that manages the frontend WebSocket connection to the cloud:
    
    ```typescript
    function useWebSocket() {
      const wsRef = useRef<WebSocket | null>(null);
      const reconnectTimeoutRef = useRef<number>();
      const [agentHealth, setAgentHealth] = useState<AgentHealth>({
        agent: false, obs: false, ptz: false
      });
      const [commandResults, setCommandResults] = useState<Map<string, CommandResult>>(new Map());
      const callbacksRef = useRef<Map<string, (result: CommandResult) => void>>(new Map());

      function connect() {
        const token = getToken();
        if (!token) return;
        
        const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
        const wsUrl = `${protocol}//${window.location.host}/frontend?token=${token}`;
        
        const ws = new WebSocket(wsUrl);
        wsRef.current = ws;
        
        ws.onmessage = (event) => {
          const msg = JSON.parse(event.data);
          if (msg.type === 'agent_health') {
            setAgentHealth({
              agent: true, // If we're receiving health, agent is connected
              obs: msg.subsystems.obs,
              ptz: msg.subsystems.ptz,
            });
          } else if (msg.type === 'command_result') {
            const callback = callbacksRef.current.get(msg.requestId);
            if (callback) {
              callback({ requestId: msg.requestId, status: msg.status, error: msg.error });
              callbacksRef.current.delete(msg.requestId);
            }
            setCommandResults(prev => new Map(prev).set(msg.requestId, { 
              requestId: msg.requestId, status: msg.status, error: msg.error 
            }));
          }
        };
        
        ws.onclose = () => {
          setAgentHealth(prev => ({ ...prev, agent: false }));
          reconnectTimeoutRef.current = setTimeout(connect, 3000);
        };
      }

      function onCommandResult(requestId: string, callback: (result: CommandResult) => void) {
        callbacksRef.current.set(requestId, callback);
      }

      useEffect(() => {
        connect();
        return () => {
          wsRef.current?.close();
          clearTimeout(reconnectTimeoutRef.current);
        };
      }, []);

      return { agentHealth, commandResults, onCommandResult };
    }
    ```
    
    The `agent` field in health is `true` whenever we receive any health message (implies tunnel is up). It goes `false` on WebSocket close.

    **Step 1 — Command State Hook (`cloud/client/src/hooks/useCommandState.ts`):**
    Implements the optimistic UI with ACK feedback pattern from CONTEXT.md:

    ```typescript
    type CommandStatus = 'idle' | 'pending' | 'ok' | 'error';

    function useCommandState() {
      const [activeScene, setActiveScene] = useState<string | null>(null);
      const [sceneStatus, setSceneStatus] = useState<CommandStatus>('idle');
      const [pendingRequestId, setPendingRequestId] = useState<string | null>(null);
      const [presetStatuses, setPresetStatuses] = useState<Map<number, CommandStatus>>(new Map());
      const { showToast } = useToast();
      const { onCommandResult } = useWebSocket();

      function executeCommand(requestId: string, action: string, onSuccess?: () => void) {
        // Latest-command-wins: replace any pending command
        setPendingRequestId(requestId);
        
        onCommandResult(requestId, (result) => {
          setPendingRequestId(null);
          if (result.status === 'ok') {
            showToast(`${action} confirmed`, 'success');
            onSuccess?.();
          } else if (result.status === 'error' || result.status === 'stale') {
            showToast(`${action} failed: ${result.error || 'unknown error'}`, 'error');
            // Flash red + revert handled in component via CSS animation
          }
        });
      }

      async function switchScene(sceneName: string) {
        setSceneStatus('pending');
        setActiveScene(null); // Optimistic clear until confirmed
        const { requestId } = await api.switchScene(sceneName);
        executeCommand(requestId, `Switch to ${sceneName}`, () => {
          setActiveScene(sceneName);
          setSceneStatus('ok');
        });
      }

      // Similar for recallPreset(presetNumber) and savePreset(presetNumber)
      // ...
      
      return { activeScene, sceneStatus, pendingRequestId, presetStatuses, switchScene, recallPreset, savePreset };
    }
    ```

    **Latest-command-wins rule:** When a new command is issued, the previous pending `requestId` is ignored. The `onCommandResult` callback is replaced — the old callback never fires. This means if the operator clicks "Preset 3" then immediately "Preset 5", only "Preset 5" resolves. Per CONTEXT.md: "most recent intent matters."
